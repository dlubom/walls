50 /*
51  * SSIM(x,y)=(2*ux*uy + C1)*(2sxy + C2) / (ux^2 + uy^2 + C1)*(sx^2 + sy^2 + C2)
52  * where,
53  * ux = SUM(w*x)
54  * sx = (SUM(w*(x-ux)^2)^0.5
55  * sxy = SUM(w*(x-ux)*(y-uy))
56  *
57  * Returns mean SSIM. MSSIM(X,Y) = 1/M * SUM(SSIM(x,y))
58  */
59 float iqa_ssim(const unsigned char *ref, const unsigned char *cmp, int w, int h, int stride,
60  int gaussian, const struct iqa_ssim_args *args)
61 {
62  int scale;
63  int x,y,src_offset,offset;
64  float *ref_f,*cmp_f;
65  struct _kernel low_pass;
66  struct _kernel window;
67  float result;
68  double ssim_sum=0.0;
69  struct _map_reduce mr;
70 
71  /* Initialize algorithm parameters */
72  scale = _max( 1, _round( (float)_min(w,h) / 256.0f ) );
73  if (args) {
74  if(args->f)
75  scale = args->f;
76  mr.map = _ssim_map;
77  mr.reduce = _ssim_reduce;
78  mr.context = (void*)&ssim_sum;
79  }
80  window.kernel = (float*)g_square_window;
81  window.w = window.h = SQUARE_LEN;
82  window.normalized = 1;
83  window.bnd_opt = KBND_SYMMETRIC;
84  if (gaussian) {
85  window.kernel = (float*)g_gaussian_window;
86  window.w = window.h = GAUSSIAN_LEN;
87  }
88 
89  /* Convert image values to floats. Forcing stride = width. */
90  ref_f = (float*)malloc(w*h*sizeof(float));
91  cmp_f = (float*)malloc(w*h*sizeof(float));
92  if (!ref_f || !cmp_f) {
93  if (ref_f) free(ref_f);
94  if (cmp_f) free(cmp_f);
95  return INFINITY;
96  }
97  for (y=0; y<h; ++y) {
98  src_offset = y*stride;
99  offset = y*w;
100  for (x=0; x<w; ++x, ++offset, ++src_offset) {
101  ref_f[offset] = (float)ref[src_offset];
102  cmp_f[offset] = (float)cmp[src_offset];
103  }
104  }
105 
106  /* Scale the images down if required */
107  if (scale > 1) {
108  /* Generate simple low-pass filter */
109  low_pass.kernel = (float*)malloc(scale*scale*sizeof(float));
110  if (!low_pass.kernel) {
111  free(ref_f);
112  free(cmp_f);
113  return INFINITY;
114  }
115  low_pass.w = low_pass.h = scale;
116  low_pass.normalized = 0;
117  low_pass.bnd_opt = KBND_SYMMETRIC;
118  for (offset=0; offset<scale*scale; ++offset)
119  low_pass.kernel[offset] = 1.0f/(scale*scale);
120 
121  /* Resample */
122  if (_iqa_decimate(ref_f, w, h, scale, &low_pass, 0, 0, 0) ||
123  _iqa_decimate(cmp_f, w, h, scale, &low_pass, 0, &w, &h)) { /* Update w/h */
124  free(ref_f);
125  free(cmp_f);
126  free(low_pass.kernel);
127  return INFINITY;
128  }
129  free(low_pass.kernel);
130  }
131 
132  result = _iqa_ssim(ref_f, cmp_f, w, h, &window, &mr, args);
133 
134  free(ref_f);
135  free(cmp_f);
136 
137  return result;
138 }

141 /* _iqa_ssim */
142 float _iqa_ssim(float *ref, float *cmp, int w, int h, const struct _kernel *k, const struct _map_reduce *mr, const struct iqa_ssim_args *args)
143 {
144  float alpha=1.0f, beta=1.0f, gamma=1.0f;
145  int L=255;
146  float K1=0.01f, K2=0.03f;
147  float C1,C2,C3;
148  int x,y,offset;
149  float *ref_mu,*cmp_mu,*ref_sigma_sqd,*cmp_sigma_sqd,*sigma_both;
150  double ssim_sum, numerator, denominator;
151  double luminance_comp, contrast_comp, structure_comp, sigma_root;
152  struct _ssim_int sint;
153 
154  /* Initialize algorithm parameters */
155  if (args) {
156  if (!mr)
157  return INFINITY;
158  alpha = args->alpha;
159  beta = args->beta;
160  gamma = args->gamma;
161  L = args->L;
162  K1 = args->K1;
163  K2 = args->K2;
164  }
165  C1 = (K1*L)*(K1*L);
166  C2 = (K2*L)*(K2*L);
167  C3 = C2 / 2.0f;
168 
169  ref_mu = (float*)malloc(w*h*sizeof(float));
170  cmp_mu = (float*)malloc(w*h*sizeof(float));
171  ref_sigma_sqd = (float*)malloc(w*h*sizeof(float));
172  cmp_sigma_sqd = (float*)malloc(w*h*sizeof(float));
173  sigma_both = (float*)malloc(w*h*sizeof(float));
174  if (!ref_mu || !cmp_mu || !ref_sigma_sqd || !cmp_sigma_sqd || !sigma_both) {
175  if (ref_mu) free(ref_mu);
176  if (cmp_mu) free(cmp_mu);
177  if (ref_sigma_sqd) free(ref_sigma_sqd);
178  if (cmp_sigma_sqd) free(cmp_sigma_sqd);
179  if (sigma_both) free(sigma_both);
180  return INFINITY;
181  }
182 
183  /* Calculate mean */
184  _iqa_convolve(ref, w, h, k, ref_mu, 0, 0);
185  _iqa_convolve(cmp, w, h, k, cmp_mu, 0, 0);
186 
187  for (y=0; y<h; ++y) {
188  offset = y*w;
189  for (x=0; x<w; ++x, ++offset) {
190  ref_sigma_sqd[offset] = ref[offset] * ref[offset];
191  cmp_sigma_sqd[offset] = cmp[offset] * cmp[offset];
192  sigma_both[offset] = ref[offset] * cmp[offset];
193  }
194  }
195 
196  /* Calculate sigma */
197  _iqa_convolve(ref_sigma_sqd, w, h, k, 0, 0, 0);
198  _iqa_convolve(cmp_sigma_sqd, w, h, k, 0, 0, 0);
199  _iqa_convolve(sigma_both, w, h, k, 0, &w, &h); /* Update the width and height */
200 
201  /* The convolution results are smaller by the kernel width and height */
202  for (y=0; y<h; ++y) {
203  offset = y*w;
204  for (x=0; x<w; ++x, ++offset) {
205  ref_sigma_sqd[offset] -= ref_mu[offset] * ref_mu[offset];
206  cmp_sigma_sqd[offset] -= cmp_mu[offset] * cmp_mu[offset];
207  sigma_both[offset] -= ref_mu[offset] * cmp_mu[offset];
208  }
209  }
210 
211  ssim_sum = 0.0;
212  for (y=0; y<h; ++y) {
213  offset = y*w;
214  for (x=0; x<w; ++x, ++offset) {
215 
216  if (!args) {
217  /* The default case */
218  numerator = (2.0 * ref_mu[offset] * cmp_mu[offset] + C1) * (2.0 * sigma_both[offset] + C2);
219  denominator = (ref_mu[offset]*ref_mu[offset] + cmp_mu[offset]*cmp_mu[offset] + C1) *
220  (ref_sigma_sqd[offset] + cmp_sigma_sqd[offset] + C2);
221  ssim_sum += numerator / denominator;
222  }
223  else {
224  /* User tweaked alpha, beta, or gamma */
225 
226  /* passing a negative number to sqrt() cause a domain error */
227  if (ref_sigma_sqd[offset] < 0.0f)
228  ref_sigma_sqd[offset] = 0.0f;
229  if (cmp_sigma_sqd[offset] < 0.0f)
230  cmp_sigma_sqd[offset] = 0.0f;
231  sigma_root = sqrt(ref_sigma_sqd[offset] * cmp_sigma_sqd[offset]);
232 
233  luminance_comp = _calc_luminance(ref_mu[offset], cmp_mu[offset], C1, alpha);
234  contrast_comp = _calc_contrast(sigma_root, ref_sigma_sqd[offset], cmp_sigma_sqd[offset], C2, beta);
235  structure_comp = _calc_structure(sigma_both[offset], sigma_root, ref_sigma_sqd[offset], cmp_sigma_sqd[offset], C3, gamma);
236 
237  sint.l = luminance_comp;
238  sint.c = contrast_comp;
239  sint.s = structure_comp;
240 
241  if (mr->map(&sint, mr->context))
242  return INFINITY;
243  }
244  }
245  }
246 
247  free(ref_mu);
248  free(cmp_mu);
249  free(ref_sigma_sqd);
250  free(cmp_sigma_sqd);
251  free(sigma_both);
252 
253  if (!args)
254  return (float)(ssim_sum / (double)(w*h));
255  return mr->reduce(w, h, mr->context);
256 }

275 /* _calc_luminance */
276 IQA_INLINE static double _calc_luminance(float mu1, float mu2, float C1, float alpha)
277 {
278  double result;
279  float sign;
280  /* For MS-SSIM* */
281  if (C1 == 0 && mu1*mu1 == 0 && mu2*mu2 == 0)
282  return 1.0;
283  result = (2.0 * mu1 * mu2 + C1) / (mu1*mu1 + mu2*mu2 + C1);
284  if (alpha == 1.0f)
285  return result;
286  sign = result < 0.0 ? -1.0f : 1.0f;
287  return sign * pow(fabs(result),(double)alpha);
288 }

290 /* _calc_contrast */
291 IQA_INLINE static double _calc_contrast(double sigma_comb_12, float sigma1_sqd, float sigma2_sqd, float C2, float beta)
292 {
293  double result;
294  float sign;
295  /* For MS-SSIM* */
296  if (C2 == 0 && sigma1_sqd + sigma2_sqd == 0)
297  return 1.0;
298  result = (2.0 * sigma_comb_12 + C2) / (sigma1_sqd + sigma2_sqd + C2);
299  if (beta == 1.0f)
300  return result;
301  sign = result < 0.0 ? -1.0f : 1.0f;
302  return sign * pow(fabs(result),(double)beta);
303 }

305 /* _calc_structure */
306 IQA_INLINE static double _calc_structure(float sigma_12, double sigma_comb_12, float sigma1, float sigma2, float C3, float gamma)
307 {
308  double result;
309  float sign;
310  /* For MS-SSIM* */
311  if (C3 == 0 && sigma_comb_12 == 0) {
312  if (sigma1 == 0 && sigma2 == 0)
313  return 1.0;
314  else if (sigma1 == 0 || sigma2 == 0)
315  return 0.0;
316  }
317  result = (sigma_12 + C3) / (sigma_comb_12 + C3);
318  if (gamma == 1.0f)
319  return result;
320  sign = result < 0.0 ? -1.0f : 1.0f;
321  return sign * pow(fabs(result),(double)gamma);
322 }
