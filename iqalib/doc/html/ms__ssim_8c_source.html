83 /* Called to calculate the final result */
84 float _ms_ssim_reduce(int w, int h, void *ctx)
85 {
86  double size = (double)(w*h);
87  struct _context *ms_ctx = (struct _context*)ctx;
88  ms_ctx->l = pow(ms_ctx->l / size, (double)ms_ctx->alpha);
89  ms_ctx->c = pow(ms_ctx->c / size, (double)ms_ctx->beta);
90  ms_ctx->s = pow(fabs(ms_ctx->s / size), (double)ms_ctx->gamma);
91  return (float)(ms_ctx->l * ms_ctx->c * ms_ctx->s);
92 }

102 /* Allocates the scaled buffers. If error, all buffers are free'd */
103 int _alloc_buffers(float **buf, int w, int h, int scales)
104 {
105  int idx;
106  int cur_w = w;
107  int cur_h = h;
108  for (idx=0; idx<scales; ++idx) {
109  buf[idx] = (float*)malloc(cur_w*cur_h*sizeof(float));
110  if (!buf[idx]) {
111  _free_buffers(buf, idx);
112  return 1;
113  }
114  cur_w = cur_w/2 + (cur_w&1);
115  cur_h = cur_h/2 + (cur_h&1);
116  }
117  return 0;
118 }

120 /*
121  * MS_SSIM(X,Y) = Lm(x,y)^aM * MULT[j=1->M]( Cj(x,y)^bj * Sj(x,y)^gj )
122  * where,
123  * L = mean
124  * C = variance
125  * S = cross-correlation
126  *
127  * b1=g1=0.0448, b2=g2=0.2856, b3=g3=0.3001, b4=g4=0.2363, a5=b5=g5=0.1333
128  */
129 float iqa_ms_ssim(const unsigned char *ref, const unsigned char *cmp, int w, int h,
130  int stride, const struct iqa_ms_ssim_args *args)
131 {
132  int wang=0;
133  int scales=SCALES;
134  int gauss=1;
135  const float *alphas=g_alphas, *betas=g_betas, *gammas=g_gammas;
136  int idx,x,y,cur_w,cur_h;
137  int offset,src_offset;
138  float **ref_imgs, **cmp_imgs; /* Array of pointers to scaled images */
139  float msssim;
140  struct _kernel lpf, window;
141  struct iqa_ssim_args s_args;
142  struct _map_reduce mr;
143  struct _context ms_ctx;
144 
145  if (args) {
146  wang = args->wang;
147  gauss = args->gaussian;
148  scales = args->scales;
149  if (args->alphas)
150  alphas = args->alphas;
151  if (args->betas)
152  betas = args->betas;
153  if (args->gammas)
154  gammas = args->gammas;
155  }
156 
157  /* Make sure we won't scale below 1x1 */
158  cur_w = w;
159  cur_h = h;
160  for (idx=0; idx<scales; ++idx) {
161  if ( gauss ? cur_w<GAUSSIAN_LEN || cur_h<GAUSSIAN_LEN : cur_w<LPF_LEN || cur_h<LPF_LEN )
162  return INFINITY;
163  cur_w /= 2;
164  cur_h /= 2;
165  }
166 
167  window.kernel = (float*)g_square_window;
168  window.w = window.h = SQUARE_LEN;
169  window.normalized = 1;
170  window.bnd_opt = KBND_SYMMETRIC;
171  if (gauss) {
172  window.kernel = (float*)g_gaussian_window;
173  window.w = window.h = GAUSSIAN_LEN;
174  }
175 
176  mr.map = _ms_ssim_map;
177  mr.reduce = _ms_ssim_reduce;
178 
179  /* Allocate the scaled image buffers */
180  ref_imgs = (float**)malloc(scales*sizeof(float*));
181  cmp_imgs = (float**)malloc(scales*sizeof(float*));
182  if (!ref_imgs || !cmp_imgs) {
183  if (ref_imgs) free(ref_imgs);
184  if (cmp_imgs) free(cmp_imgs);
185  return INFINITY;
186  }
187  if (_alloc_buffers(ref_imgs, w, h, scales)) {
188  free(ref_imgs);
189  free(cmp_imgs);
190  return INFINITY;
191  }
192  if (_alloc_buffers(cmp_imgs, w, h, scales)) {
193  _free_buffers(ref_imgs, scales);
194  free(ref_imgs);
195  free(cmp_imgs);
196  return INFINITY;
197  }
198 
199  /* Copy original images into first scale buffer, forcing stride = width. */
200  for (y=0; y<h; ++y) {
201  src_offset = y*stride;
202  offset = y*w;
203  for (x=0; x<w; ++x, ++offset, ++src_offset) {
204  ref_imgs[0][offset] = (float)ref[src_offset];
205  cmp_imgs[0][offset] = (float)cmp[src_offset];
206  }
207  }
208 
209  /* Create scaled versions of the images */
210  cur_w=w;
211  cur_h=h;
212  lpf.kernel = (float*)g_lpf;
213  lpf.w = lpf.h = LPF_LEN;
214  lpf.normalized = 1;
215  lpf.bnd_opt = KBND_SYMMETRIC;
216  for (idx=1; idx<scales; ++idx) {
217  if (_iqa_decimate(ref_imgs[idx-1], cur_w, cur_h, 2, &lpf, ref_imgs[idx], 0, 0) ||
218  _iqa_decimate(cmp_imgs[idx-1], cur_w, cur_h, 2, &lpf, cmp_imgs[idx], &cur_w, &cur_h))
219  {
220  _free_buffers(ref_imgs, scales);
221  _free_buffers(cmp_imgs, scales);
222  free(ref_imgs);
223  free(cmp_imgs);
224  return INFINITY;
225  }
226  }
227 
228  cur_w=w;
229  cur_h=h;
230  msssim = 1.0;
231  for (idx=0; idx<scales; ++idx) {
232 
233  ms_ctx.l = 0;
234  ms_ctx.c = 0;
235  ms_ctx.s = 0;
236  ms_ctx.alpha = alphas[idx];
237  ms_ctx.beta = betas[idx];
238  ms_ctx.gamma = gammas[idx];
239 
240  if (!wang) {
241  /* MS-SSIM* (Rouse/Hemami) */
242  s_args.alpha = 1.0f;
243  s_args.beta = 1.0f;
244  s_args.gamma = 1.0f;
245  s_args.K1 = 0.0f; /* Force stabilization constants to 0 */
246  s_args.K2 = 0.0f;
247  s_args.L = 255;
248  s_args.f = 1; /* Don't resize */
249  mr.context = &ms_ctx;
250  msssim *= _iqa_ssim(ref_imgs[idx], cmp_imgs[idx], cur_w, cur_h, &window, &mr, &s_args);
251  }
252  else {
253  /* MS-SSIM (Wang) */
254  s_args.alpha = 1.0f;
255  s_args.beta = 1.0f;
256  s_args.gamma = 1.0f;
257  s_args.K1 = 0.01f;
258  s_args.K2 = 0.03f;
259  s_args.L = 255;
260  s_args.f = 1; /* Don't resize */
261  mr.context = &ms_ctx;
262  msssim *= _iqa_ssim(ref_imgs[idx], cmp_imgs[idx], cur_w, cur_h, &window, &mr, &s_args);
263  }
264 
265  if (msssim == INFINITY)
266  break;
267  cur_w = cur_w/2 + (cur_w&1);
268  cur_h = cur_h/2 + (cur_h&1);
269  }
270 
271  _free_buffers(ref_imgs, scales);
272  _free_buffers(cmp_imgs, scales);
273  free(ref_imgs);
274  free(cmp_imgs);
275 
276  return msssim;
277 }
